111def convert_to_prices( 

112 param: ModelParameters, log_returns: np.ndarray 

113) -> np.ndarray: 

114 """ 

115 This method converts a sequence of log returns into normal returns 

116 (exponentiation) and then computes a price 

117 sequence given a starting price, param.all_s0. 

118 :param param: the model parameters object 

119 :param log_returns: the log returns to exponentiated 

120 :return: 

121 """ 

122 assert isinstance( 

123 param, ModelParameters 

124 ), "param must be an object of Class ModelParameters" 

125 returns = convert_to_returns(log_returns) 

126 # A sequence of prices starting with param.all_s0 

127 price_sequence: list = [param.all_s0] 

128 for n in range(1, len(returns)): 

129 # Add the price at t-1 * return at t 

130 price_sequence.append(price_sequence[n - 1] * returns[n - 1]) 

131 return np.array(price_sequence)

185def brownian_motion_log_returns( 

186 param: ModelParameters, seed: int = None 

187) -> np.ndarray: 

188 """ 

189 This method returns a Wiener process. The Wiener process is also called 

190 Brownian motion. For more information about the Wiener process check out 

191 the Wikipedia page: http://en.wikipedia.org/wiki/Wiener_process 

192 :param param: the model parameters object 

193 :param seed: 

194 :return: brownian motion log returns 

195 """ 

196 assert isinstance( 

197 param, ModelParameters 

198 ), "param must be an object of Class ModelParameters" 

199 if seed is not None: 

200 nrand.seed(seed) 

201 sqrt_delta_sigma = math.sqrt(param.all_delta) * param.all_sigma 

202 return nrand.normal(loc=0, scale=sqrt_delta_sigma, size=param.all_time)

223def geometric_brownian_motion_log_returns( 

224 param: ModelParameters, seed: int = None 

225) -> np.ndarray: 

226 """ 

227 This method constructs a sequence of log returns which, when 

228 exponentiated, produce a random Geometric Brownian Motion (GBM). 

229 GBM is the stochastic process underlying the Black Scholes 

230 options pricing formula. 

231 :param param: model parameters object 

232 :param seed: 

233 :return: returns the log returns of a geometric brownian motion process 

234 """ 

235 assert isinstance( 

236 param, ModelParameters 

237 ), "param must be an object of Class ModelParameters" 

238 wiener_process = np.array(brownian_motion_log_returns(param, seed=seed)) 

239 sigma_pow_mu_delta = ( 

240 param.gbm_mu - 0.5 * math.pow(param.all_sigma, 2.0) 

241 ) * param.all_delta 

242 return wiener_process + sigma_pow_mu_delta

263def jump_diffusion_process(param: ModelParameters, seed: int = None) -> list: 

264 """ 

265 This method produces a sequence of Jump Sizes which represent a jump 

266 diffusion process. These jumps are combined with a geometric brownian 

267 motion (log returns) to produce the Merton model. 

268 :param param: the model parameters object 

269 :param seed: 

270 :return: jump sizes for each point in time 

271 (mostly zeroes if jumps are infrequent) 

272 """ 

273 assert isinstance( 

274 param, ModelParameters 

275 ), "param must be an object of Class ModelParameters" 

276 if seed is not None: 

277 nrand.seed(seed) 

278 s_n = time = 0 

279 small_lamda = -(1.0 / param.lamda) 

280 jump_sizes = [] 

281 for k in range(0, param.all_time): 

282 jump_sizes.append(0.0) 

283 while s_n < param.all_time: 

284 s_n += small_lamda * math.log(nrand.uniform(0, 1)) 

285 for j in range(0, param.all_time): 

286 if ( 

287 time * param.all_delta 

288 <= s_n * param.all_delta 

289 <= (j + 1) * param.all_delta 

290 ): 

291 jump_sizes[j] += nrand.normal( 

292 param.jumps_mu, param.jumps_sigma 

293 ) 

294 break 

295 time += 1 

296 return jump_sizes

340def heston_construct_correlated_path( 

341 param: ModelParameters, brownian_motion_one: np.ndarray, seed: int = None 

342) -> Tuple[np.ndarray, np.ndarray]: 

343 """ 

344 This method is a simplified version of the Cholesky decomposition method 

345 for just two assets. It does not make use of matrix algebra and is 

346 therefore quite easy to implement. 

347 :param param: model parameters object 

348 :param brownian_motion_one: A first path to correlate against 

349 :param seed: 

350 :return: a correlated brownian motion path 

351 """ 

352 assert isinstance( 

353 param, ModelParameters 

354 ), "param must be an object of Class ModelParameters" 

355 if seed is not None: 

356 nrand.seed(seed) 

357 # We do not multiply by sigma here, we do that in the Heston model 

358 sqrt_delta = math.sqrt(param.all_delta) 

359 # Construct a path correlated to the first path 

360 brownian_motion_two = [] 

361 for n in range(param.all_time - 1): 

362 term_one = param.cir_rho * brownian_motion_one[n] 

363 term_two = math.sqrt(1 - math.pow(param.cir_rho, 2.0)) * nrand.normal( 

364 0, sqrt_delta 

365 ) 

366 brownian_motion_two.append(term_one + term_two) 

367 return np.array(brownian_motion_one), np.array(brownian_motion_two)

370def cox_ingersoll_ross_heston( 

371 param: ModelParameters, seed: int = None 

372) -> Tuple[np.ndarray, np.ndarray]: 

373 """ 

374 This method returns the rate levels of a mean-reverting Cox Ingersoll Ross 

375 process. It is used to model interest rates as well as stochastic 

376 volatility in the Heston model. Because the returns between the underlying 

377 and the stochastic volatility should be correlated we pass a correlated 

378 Brownian motion process into the method from which the interest rate levels 

379 are constructed. The other correlated process is used in the Heston model 

380 :param param: the model parameters objects 

381 :param seed: 

382 :return: the interest rate levels for the CIR process 

383 """ 

384 assert isinstance( 

385 param, ModelParameters 

386 ), "param must be an object of Class ModelParameters" 

387 if seed is not None: 

388 nrand.seed(seed) 

389 # We don't multiply by sigma here because we do that in heston 

390 sqrt_delta_sigma = math.sqrt(param.all_delta) * param.all_sigma 

391 brownian_motion_volatility = nrand.normal( 

392 loc=0, scale=sqrt_delta_sigma, size=param.all_time 

393 ) 

394 a, mu, zero = param.heston_a, param.heston_mu, param.heston_vol0 

395 volatilities = [zero] 

396 for h in range(1, param.all_time): 

397 drift = a * (mu - volatilities[-1]) * param.all_delta 

398 randomness = ( 

399 math.sqrt(max(volatilities[h - 1], 0.05)) 

400 * brownian_motion_volatility[h - 1] 

401 ) 

402 volatilities.append(max(volatilities[-1], 0.05) + drift + randomness) 

403 return np.array(brownian_motion_volatility), np.array(volatilities)

449def cox_ingersoll_ross_levels( 

450 param: ModelParameters, seed: int = None 

451) -> np.ndarray: 

452 """ 

453 This method returns the rate levels of a mean-reverting Cox Ingersoll Ross 

454 process. It is used to model interest rates as well as stochastic 

455 volatility in the Heston model. Because the returns between the underlying 

456 and the stochastic volatility should be correlated we pass a correlated 

457 Brownian motion process into the method from which the interest rate levels 

458 are constructed. The other correlated process is used in the Heston model 

459 :param param: the model parameters object 

460 :param seed: 

461 :return: the interest rate levels for the CIR process 

462 """ 

463 assert isinstance( 

464 param, ModelParameters 

465 ), "param must be an object of Class ModelParameters" 

466 brownian_motion = brownian_motion_log_returns(param, seed=seed) 

467 # Setup the parameters for interest rates 

468 a, mu, zero = param.cir_a, param.cir_mu, param.all_r0 

469 # Assumes output is in levels 

470 levels = [zero] 

471 for h in range(1, param.all_time): 

472 drift = a * (mu - levels[h - 1]) * param.all_delta 

473 """ The main difference between this and the Ornstein Uhlenbeck model  

474 is that we multiply the 'random' component by the square-root of the  

475 previous level i.e. the process has level dependent interest  

476 rates. """ 

477 randomness = math.sqrt(levels[h - 1]) * brownian_motion[h - 1] 

478 levels.append(levels[h - 1] + drift + randomness) 

479 return np.array(levels)

482def ornstein_uhlenbeck_levels( 

483 param: ModelParameters, seed: int = None 

484) -> list: 

485 """ 

486 This method returns the rate levels of a mean-reverting 

487 Ornstein Uhlenbeck process. 

488 :param param: the model parameters object 

489 :param seed: 

490 :return: the interest rate levels for the Ornstein Uhlenbeck process 

491 """ 

492 assert isinstance( 

493 param, ModelParameters 

494 ), "param must be an object of Class ModelParameters" 

495 ou_levels = [param.all_r0] 

496 brownian_motion_returns = brownian_motion_log_returns(param, seed=seed) 

497 for h in range(1, param.all_time): 

498 drift = param.ou_a * (param.ou_mu - ou_levels[h - 1]) * param.all_delta 

499 randomness = brownian_motion_returns[h - 1] 

500 ou_levels.append(ou_levels[h - 1] + drift + randomness) 

501 return ou_levels
